# Solicitar peso e altura do usuário
peso = float (input("30 kg"))
altura = float (input("1,80))

# Calcular IMC
imc = peso/(altura**2)

# Determina a faixa do IMC
if imc < 18.5:
faixa = "Abaixo do peso"
elif < 24.9
faixa = "peso normal"
elif < 29.9 
faixa = "Sobrepeso"
elif < 34.9
faixa = "Obesidade grau 1 "
elif< 39.9 
faixa = "Obesidade grau 2"
else:
    faixa = "Obesidade grau 3"

#Exibir resultado
print (f "S IMC é:{imc:.sf}")
print (f"você esta na faixa:{faixa}")
